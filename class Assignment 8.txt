package execution;

import java.io.File;
import java.io.IOException;
import java.time.Duration;
import java.util.List;

import org.openqa.selenium.By;
import org.openqa.selenium.JavascriptExecutor;
import org.openqa.selenium.OutputType;
import org.openqa.selenium.TakesScreenshot;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.interactions.Actions;
import org.openqa.selenium.io.FileHandler;

public class ClassAssignment8 {
	public static void main(String[] args) throws IOException {
		WebDriver driver= new ChromeDriver();
		driver.manage().window().maximize();
		//Enter cricbuzz url
		driver.get("https://www.cricbuzz.com/");
		driver.manage().timeouts().implicitlyWait(Duration.ofSeconds(20));
		
		//mouse Hover on more
		Actions act= new Actions(driver);
		WebElement more = driver.findElement(By.xpath("//a[text()='More']"));
		act.moveToElement(more).build().perform();
		
		//click on Careers option
		driver.findElement(By.xpath("//a[@title='Careers']")).click();
		
		//Click on "Database Administrator"
		driver.findElement(By.linkText("Database Administrator")).click();
		
		//Scroll till submit 
		WebElement cap = driver.findElement(By.xpath("//button[text()='Submit']"));
		JavascriptExecutor js= (JavascriptExecutor)driver;
		js.executeScript("arguments[0].scrollIntoView(true)",cap);
		
		//Enter the deatils
		driver.findElement(By.xpath("//input[@placeholder='Enter the name']")).sendKeys("Dingi");
		driver.findElement(By.xpath("//input[@placeholder='Enter your email Id']")).sendKeys("Dingi@gmail.com");
		driver.findElement(By.xpath("//textarea[@placeholder='Describe yourself']")).sendKeys("abcdefghijklmnopqrstuvwxyz");
		driver.findElement(By.xpath("//input[@placeholder='Link 1']")).sendKeys("https://github.com/dingi12");
		
		File f=new File("C:\\Users\\ASUS\\Downloads\\NameDIKSHA VG 8951056631 (1) (1).pdf");
		 String resume = f.getAbsolutePath();
		driver.findElement(By.xpath("//input[@type='file']")).sendKeys(resume);
		
		//driver.findElement(By.xpath("//div[@role='presentation']")).click();
		
		//click on submit
		driver.findElement(By.xpath("//button[text()='Submit']")).click();
		
		//Take a Screeshot of webpage
		TakesScreenshot ts=(TakesScreenshot)driver;
		File temp = ts.getScreenshotAs(OutputType.FILE);
		File dest=new File("./Screenshot/image8.png");
		FileHandler.copy(temp, dest);
		
		//Scroll till bottom 
		js.executeScript("(window.scrollTo(0,document.body.scrollHeight))");
		
		//click on Twitter
		driver.findElement(By.xpath("//a[text()='Twitter']")).click();
		
		//print the count of images present on webpage
		 List<WebElement> img = driver.findElements(By.tagName("img"));
