package assignment;

import java.io.File;
import java.io.IOException;

import org.openqa.selenium.By;
import org.openqa.selenium.OutputType;
import org.openqa.selenium.TakesScreenshot;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.io.FileHandler;

public class petdiseasealert {
	public static void main(String[] args) throws IOException, InterruptedException {
		WebDriver driver = new ChromeDriver();
		driver.manage().window().maximize();
		driver.get("https://petdiseasealerts.org/");
		
		//singup isDisplayed or not
		boolean status = driver.findElement(By.xpath("//a[text()='Sign up']")).isDisplayed();
		System.out.println(status);
		
		//click on that button
		driver.findElement(By.xpath("//a[text()='Sign up']")).click();
		
		//click on agreement and verify it is selected or not
		boolean newstatus = driver.findElement(By.xpath("//input[@type='checkbox']")).isSelected();
		System.out.println(newstatus);
		
		
		Thread.sleep(2000);
		//to verify the signup is enabled or not
		boolean enablestatus = driver.findElement(By.xpath("//button[text()='Sign Up']")).isEnabled();
		System.out.println(enablestatus);
	

		//Take screenshot of webpage
		
		TakesScreenshot ts=(TakesScreenshot)driver;
		File temp = ts.getScreenshotAs(OutputType.FILE);
		File dest=new File("./Screenshot/image4.png");
		FileHandler.copy(temp, dest);
		
	}

}
